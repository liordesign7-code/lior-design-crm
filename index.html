function sendLead() { const name = document.getElementById("name").value; const phone = document.getElementById("phone").value; const product = document.getElementById("product").value; fetch("https://lior-design-crm-api.onrender.com/api/leads", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name: name, phone: phone, product: product }) }) .then(response => response.json()) .then(data => { alert("הליד נשמר במערכת בהצלחה"); }) .catch(error => { alert("שגיאה בשליחת הליד"); console.error(error); }); }
